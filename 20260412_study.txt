<script setup>
import { ref } from "vue";

const ROW_COUNT = 5;
const ROW_LABEL_WIDTH = "2.5rem";

// 列定義（上から順に表示 / expanded: true = 開いている）
const columnDatas = ref([
    {
        key: "issue",
        longLabel: "解決すべき課題",
        shortLabel: "課題",
        collapsedWidth: "1.75rem",
        expanded: true,
    },
    {
        key: "period",
        longLabel: "計画における期間aaaaaaaaaaaaaaaa",
        shortLabel: "期間",
        collapsedWidth: "1.75rem",
        expanded: true,
    },
    {
        key: "goal",
        longLabel: "達成したい目標",
        shortLabel: "目標",
        collapsedWidth: "1.75rem",
        expanded: true,
    },
    {
        key: "support",
        longLabel: "支援の内容",
        shortLabel: "支援",
        collapsedWidth: "1.75rem",
        expanded: true,
    },
    {
        key: "method",
        longLabel: "支援の方法",
        shortLabel: "方法",
        collapsedWidth: "1.75rem",
        expanded: true,
    },
    {
        key: "frequency",
        longLabel: "支援の頻度",
        shortLabel: "頻度",
        collapsedWidth: "1.75rem",
        expanded: true,
    },
    {
        key: "evaluation",
        longLabel: "評価の方法",
        shortLabel: "評価",
        collapsedWidth: "1.75rem",
        expanded: true,
    },
    {
        key: "remark",
        longLabel: "備考・特記事項",
        shortLabel: "備考",
        collapsedWidth: "2rem",
        expanded: true,
    },
]);

const COL_COUNT = columnDatas.value.length;

// 行番号列は常に固定幅
const rowNumberColumnStyle = {
    flex: `0 0 ${ROW_LABEL_WIDTH}`,
    width: ROW_LABEL_WIDTH,
};

const cells = ref(
    Array.from({ length: ROW_COUNT }, () =>
        Array.from({ length: COL_COUNT }, () => ""),
    ),
);

const rowClass = "flex w-full";
const transitionClass = "transition-[width,flex] duration-300 ease-in-out";

const fieldClass =
    "block min-h-[5rem] w-full resize-y rounded-md border-0 bg-transparent px-2 py-1.5 text-sm leading-relaxed text-gray-800 placeholder:text-gray-400 focus:bg-white focus:outline-none focus:ring-2 focus:ring-indigo-400";

const cellBorderClass = "border-b border-r border-gray-300";

const labelButtonClass =
    "m-0 grid h-10 w-full grid-cols-1 grid-rows-1 overflow-hidden p-0 text-xs font-semibold leading-none transition-colors duration-300 hover:bg-gray-100";

const labelTextClass =
    "col-start-1 row-start-1 flex h-10 w-full min-w-0 items-center justify-center px-1 leading-none transition-opacity duration-300 ease-in-out";

function columnStyle(colData, expanded) {
    return expanded
        ? { flex: "1 1 0%", minWidth: "0px" }
        : {
              flex: `0 0 ${colData.collapsedWidth}`,
              width: colData.collapsedWidth,
          };
}

function toggleColumn(colIndex) {
    columnDatas.value[colIndex].expanded = !columnDatas.value[colIndex].expanded;
}
</script>

<template>
    <div
        class="overflow-hidden rounded-lg border border-gray-200 bg-white shadow-sm"
    >
        <div class="border-b border-gray-200 bg-gray-50 px-4 py-3">
            <h3 class="text-sm font-semibold text-gray-800">
                相談計画入力（練習）
            </h3>
            <p class="mt-1 text-xs text-gray-500">
                {{ ROW_COUNT }}行 ×
                {{ COL_COUNT }}列のグリッド（列ラベルをクリックで幅を切り替え）
            </p>
        </div>

        <div class="overflow-x-auto p-4">
            <div
                class="min-w-[48rem] overflow-hidden rounded-lg border border-gray-300 text-sm"
            >
                <div :class="[rowClass, 'bg-gray-50']">
                    <div
                        :class="[cellBorderClass, 'h-10 shrink-0']"
                        :style="rowNumberColumnStyle"
                    >
                        <div
                            class="flex h-10 items-center justify-center text-xs font-semibold text-gray-600"
                        >
                            #
                        </div>
                    </div>

                    <div
                        v-for="(colData, colIndex) in columnDatas"
                        :key="colData.key"
                        :class="[
                            cellBorderClass,
                            transitionClass,
                            'h-10 shrink-0 overflow-hidden',
                            colIndex === columnDatas.length - 1 ? 'border-r-0' : '',
                        ]"
                        :style="columnStyle(colData, colData.expanded)"
                    >
                        <button
                            type="button"
                            :class="[
                                labelButtonClass,
                                colData.expanded
                                    ? 'text-gray-700'
                                    : 'bg-gray-200 text-gray-500',
                            ]"
                            @click="toggleColumn(colIndex)"
                        >
                            <span
                                :class="[
                                    labelTextClass,
                                    colData.expanded
                                        ? 'opacity-100'
                                        : 'pointer-events-none opacity-0',
                                ]"
                            >
                                <span
                                    class="w-full min-w-0 truncate text-center leading-none"
                                >
                                    {{ colData.longLabel }}
                                </span>
                            </span>
                            <span
                                :class="[
                                    labelTextClass,
                                    colData.expanded
                                        ? 'pointer-events-none opacity-0'
                                        : 'opacity-100',
                                ]"
                            >
                                {{ colData.shortLabel }}
                            </span>
                        </button>
                    </div>
                </div>

                <div
                    v-for="(row, rowIndex) in cells"
                    :key="`row-${rowIndex}`"
                    :class="[
                        rowClass,
                        rowIndex === cells.length - 1
                            ? '[&>div]:border-b-0'
                            : '',
                    ]"
                >
                    <div
                        :class="[cellBorderClass, 'shrink-0 bg-gray-50']"
                        :style="rowNumberColumnStyle"
                    >
                        <div
                            class="flex h-full items-center justify-center px-1 py-2 text-xs font-medium text-gray-600"
                        >
                            {{ rowIndex + 1 }}
                        </div>
                    </div>

                    <div
                        v-for="(colData, colIndex) in columnDatas"
                        :key="`cell-${rowIndex}-${colData.key}`"
                        :class="[
                            cellBorderClass,
                            transitionClass,
                            'shrink-0 overflow-hidden p-1',
                            colIndex === columnDatas.length - 1 ? 'border-r-0' : '',
                            !colData.expanded ? 'bg-gray-50/60' : '',
                        ]"
                        :style="columnStyle(colData, colData.expanded)"
                    >
                        <textarea
                            v-model="cells[rowIndex][colIndex]"
                            rows="3"
                            :class="[
                                fieldClass,
                                'transition-opacity duration-300 ease-in-out',
                                colData.expanded
                                    ? 'opacity-100'
                                    : 'pointer-events-none opacity-0',
                            ]"
                            :placeholder="`${rowIndex + 1}-${colData.shortLabel}`"
                            :tabindex="colData.expanded ? 0 : -1"
                        />
                    </div>
                </div>
            </div>
        </div>
    </div>
</template>
