<x-app-layout>
    {{--
        【このブロック（Alpine x-data）がやっていること】

        1. なぜ数値が必要か
           CSS の position:sticky は「張り付く位置」を top で決める。
           その top は「スクロール領域の上端からの距離」であり、
           画面上でナビや緑帯の“すぐ下”に黄パネルを置きたいときは、
           （ナビの実高さ）+（緑帯の実高さ）を px で指定するのがわかりやすい。

        2. navHeightPx / topPx の役割
           - navHeightPx … layouts/navigation の <nav> の offsetHeight（sticky の緑帯の top にも使う）
           - topPx       … 黄パネル用。ナビ + 緑帯の合計高さ = sticky で「緑の真下」に黄を止めるための値

        3. ResizeObserver を使う理由
           ナビの高さ（メニュー折り返し・フォント）や緑帯の文案変更で高さが変わっても、
           再計算しないと top がズレる。Mutation のたびに測り直すのが ResizeObserver。
           ※ ウィンドウ幅だけなら resize イベントでも足りることはあるが、要素“内部”の高さ変化は ResizeObserver が向いている。

        4. init() と $nextTick
           Alpine が x-ref を束ね終わった後に計測したいので $nextTick で 1 ティック遅らせる。
           （ref がまだ取れないタイミングでの offsetHeight 0 回避）

        5. _ro を disconnect している理由
           Livewire 等でパーシャル再描画すると init が再度走ることがあり、そのたびに observe が二重になると
           コールバックが複数回登録される。再セットアップ前に古い ResizeObserver を切る。

        6. Alpine.data を別ファイルにしない理由
           alpine:init の登録順と、この Blade が読み込まれる順で「未定義」になり得るため、
           x-data のオブジェクトに init() を直書きして読み込み順に依存しないようにしている。

        【対応するコード位置の目安】
           - topPx / navHeightPx / _ro … 下の x-data 直後のプロパティ
           - init / $nextTick … DOM 確定後に計測するための待ち
           - recalc … nav と緑帯の offsetHeight を読む本体
           - ResizeObserver … nav・緑の「高さ変化」を監視
           - window.resize … ウィンドウ幅変更でナビが折り返すケースの保険（RO と併用）
           - bindScheduleScrollSync … 曜日ヘッダーとグリッド本体の横スクロールを同じ scrollLeft に揃える
    --}}
    <div class="space-y-0" x-data="{
        topPx: 0,
        navHeightPx: 0,
        _ro: null,
        _scrollSyncing: false,
        bindScheduleScrollSync() {
            const head = this.$refs.scheduleHeadScroll;
            const body = this.$refs.scheduleBodyScroll;
            if (!head || !body) return;
            const sync = (src, dst) => {
                if (this._scrollSyncing) return;
                if (dst.scrollLeft === src.scrollLeft) return;
                this._scrollSyncing = true;
                dst.scrollLeft = src.scrollLeft;
                this._scrollSyncing = false;
            };
            head.addEventListener('scroll', () => sync(head, body), { passive: true });
            body.addEventListener('scroll', () => sync(body, head), { passive: true });
        },
        init() {
            /* x-ref 確定後に offsetHeight を取る（上の Blade コメント 4） */
            this.$nextTick(() => {
                /* ナビ: layouts/navigation の先頭 nav。複数あるサイトでは querySelector を絞る */
                const nav = document.querySelector('nav');
                /* 緑帯: このコンポーネント内の x-ref（Alpine が束ねた後に参照可能） */
                const green = this.$refs.greenStrip;
    
                const recalc = () => {
                    const navH = nav ? nav.offsetHeight : 0;
                    const greenH = green ? green.offsetHeight : 0;
                    this.navHeightPx = Math.round(navH);
                    /* 黄パネル sticky の top = ナビ下端までの距離 + 緑帯の高さ（＝緑の直下に黄を合わせる） */
                    this.topPx = this.navHeightPx + Math.round(greenH);
                };
    
                recalc();
    
                if (this._ro) {
                    this._ro.disconnect();
                }
    
                /* 高さが変わったら recalc。disconnect は二重 observe 防止（上の Blade コメント 5 参照） */
                this._ro = new ResizeObserver(() => recalc());
                if (nav) {
                    this._ro.observe(nav);
                }
                if (green) {
                    this._ro.observe(green);
                }
    
                /* 幅変更でナビが折り返すときの再計算（RO と併用。Blade コメント 3 の補足） */
                window.addEventListener('resize', recalc);

                /* 曜日行は overflow-x の外で sticky（topPx がビューポート基準になる）。ヘッダーと本体の横スクロール同期 */
                this.bindScheduleScrollSync();
            });
        },
    }">
        <div x-ref="greenStrip" class="sticky z-40 border-b border-green-200 bg-green-100 px-4 py-3 shadow-sm"
            :style="`top: ${navHeightPx}px`">
            緑の固定帯（高さが変わっても ResizeObserver で追従）
        </div>
        <div class=" h-[200px] bg-pink-100"> ここにはスクロール時に消えてもいいやつが入る</div>
        {{--
            レイアウト: 黄パネルは幅固定。左列は flex-col。
            曜日ヘッダーは overflow-x-auto の外で sticky + topPx（中に入れると top がスクロール枠基準でズレる）。
            ヘッダー用と本体用で横スクロールを二段にし、bindScheduleScrollSync で scrollLeft を同期。
            本体側は時間列 sticky left + 7×24 グリッド（合計高さ 1288 - ヘッダー 40）。
        --}}
        <div class="mx-auto flex w-full max-w-[1200px] items-start gap-2 bg-red-100 px-2 sm:px-4">
            <div class="flex min-h-0 min-w-0 flex-1 flex-col">
                @php
                    /* 週グリッド検証用（ISO 週の並び: 月〜日） */
                    $weekdayLabels = ['月', '火', '水', '木', '金', '土', '日'];
                @endphp
                <div
                    class="rounded-sm border border-slate-400 bg-white text-xs text-slate-800 shadow-sm">
                    {{-- 曜日: ビューポート基準の縦 sticky（overflow-x は子のみ） --}}
                    <div class="sticky z-[35] shrink-0 bg-slate-100 shadow-sm" :style="`top: ${topPx}px`">
                        <div x-ref="scheduleHeadScroll" class="overflow-x-auto">
                            <div
                                class="flex h-10 w-[960px] shrink-0 items-stretch border-b border-slate-300 font-medium">
                                <div
                                    class="sticky left-0 z-[36] flex w-14 shrink-0 items-center border-r border-slate-300 bg-slate-100 shadow-[4px_0_6px_-2px_rgba(15,23,42,0.12)]">
                                </div>
                                <div class="flex w-[904px] shrink-0">
                                    @foreach ($weekdayLabels as $label)
                                        <div
                                            class="flex flex-1 items-center justify-center border-r border-slate-300 last:border-r-0">
                                            {{ $label }}
                                        </div>
                                    @endforeach
                                </div>
                            </div>
                        </div>
                    </div>
                    {{-- グリッド本体: 横スクロールはヘッダーと同期。スクロールバーは上（曜日）のみ表示（下は非表示で操作は同じ） --}}
                    <div
                        x-ref="scheduleBodyScroll"
                        class="overflow-x-auto [scrollbar-width:none] [-ms-overflow-style:none] [&::-webkit-scrollbar]:[display:none]">
                        <div class="inline-flex h-[1248px] w-[960px] shrink-0 flex-col">
                            <div class="flex min-h-0 flex-1">
                                <div
                                    class="sticky left-0 z-20 flex w-14 shrink-0 flex-col border-r border-slate-300 bg-slate-50 text-[11px] text-slate-600 shadow-[4px_0_8px_-2px_rgba(15,23,42,0.15)]">
                                    @for ($h = 0; $h < 24; $h++)
                                        <div
                                            class="flex flex-1 items-start justify-end border-b border-slate-200 py-0.5 pr-1 leading-none">
                                            {{ sprintf('%d:00', $h) }}
                                        </div>
                                    @endfor
                                </div>
                                <div class="grid h-full w-[904px] shrink-0 grid-cols-7 border-slate-200"
                                    style="grid-template-rows: repeat(24, minmax(0, 1fr));">
                                    @for ($h = 0; $h < 24; $h++)
                                        @for ($c = 0; $c < 7; $c++)
                                            <div
                                                class="box-border min-h-0 border-b border-r border-slate-200 bg-blue-50/40 [&:nth-child(7n)]:border-r-0">
                                            </div>
                                        @endfor
                                    @endfor
                                </div>
                            </div>
                        </div>
                    </div>
                </div>
            </div>

            <div class="w-[300px] shrink-0 self-start bg-yellow-100 shadow"
                :style="`position: sticky; top: ${topPx}px;`">
                <div class="p-4">
                    <p class="text-sm font-medium text-gray-700">操作パネル</p>
                    <p class="mt-2 text-xs text-gray-500">
                        sticky の top = ナビ + 緑帯（<span x-text="topPx"></span> px）
                    </p>
                </div>
            </div>
        </div>
    </div>
</x-app-layout>
